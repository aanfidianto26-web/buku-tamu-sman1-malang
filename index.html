        function loginAdmin() {

            const password = document.getElementById('adminPassword').value;

           

            if (password === ADMIN_PASSWORD) {

        }

        function exportToCSV() {

            if (!isAdminLoggedIn) {

            }

            }

            const headers = ['No', 'Nama Lengkap', 'No. Telepon', 'Instansi/Organisasi', 'Keperluan', 'Keterangan', 'Waktu Kunjungan', 'Tanggal'];

            const today = new Date().toLocaleDateString('id-ID');

           

            let csvContent = headers.join(',') + '\n';

           

            daftarTamuHariIni.forEach((tamu, index) => {

                const row = [

                    `"${tamu.instansi || '-'}"`,

                    `"${tamu.keperluan}"`,

                    `"${tamu.keterangan || '-'}"`,

                    `"${tamu.waktu}"`,

            });

            showCSVModal(csvContent, today);

        }

        function copyCSVToClipboard() {

            const csvData = document.getElementById('csvData');

            csvData.select();

            csvData.setSelectionRange(0, 99999); // For mobile devices

           

            try {

                document.execCommand('copy');

                showNotification('Data berhasil disalin ke clipboard!', 'success');

            } catch (err) {

                // Fallback for modern browsers

                if (navigator.clipboard) {

                    navigator.clipboard.writeText(window.currentCSVContent).then(() => {

                        showNotification('Data berhasil disalin ke clipboard!', 'success');

                    }).catch(() => {

                        showNotification('Gagal menyalin. Silakan copy manual.', 'error');

                    });

                } else {

                    showNotification('Silakan pilih semua teks dan copy manual (Ctrl+C)', 'error');

                }

            }

        }
